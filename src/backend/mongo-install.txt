# ============ RHEL ============ #
# Prep repo
touch /etc/yum.repos.d/mongodb-org-8.0.repo
echo "name=MongoDB Repository" >> /etc/yum.repos.d/mongodb-org-8.0.repo
echo "baseurl=https://repo.mongodb.org/yum/redhat/9/mongodb-org/8.0/x86_64/" >> /etc/yum.repos.d/mongodb-org-8.0.repo
echo "gpgcheck=1" >> /etc/yum.repos.d/mongodb-org-8.0.repo
echo "enabled=1" >> /etc/yum.repos.d/mongodb-org-8.0.repo
echo "gpgkey=https://pgp.mongodb.com/server-8.0.asc" >> /etc/yum.repos.d/mongodb-org-8.0.repo

# Install mongodb-org
sudo yum install mongodb-mongosh-shared-openssl3
sudo yum install mongodb-org

# Start mongodb-service
sudo systemctl start mongod
sudo systemctl status mongod

# Launch mongod
$ mongosh

# Connecting with Node.js
$ npm install mongodb

# ============ Windows ============ #
# Download installer 
https://fastdl.mongodb.org/windows/mongodb-windows-x86_64-8.0.3-signed.msi

# Follow install prompt

# Create new DB folder in desired location

# Open Command Prompt as admin and paste:
"C:\Program Files\MongoDB\Server\8.0\bin\mongod.exe" --dbpath="c:[PATH-TO-DB]"
	- optionally, add to mongod.exe binary to $PATH

# Open MongoDBCompass app

# Click the Add New Connection button, or the plus in the Connections column

# enter localhost:27017 into URI

# Select Save & Connect

# install npm mongodb tools
npm install mongodb
